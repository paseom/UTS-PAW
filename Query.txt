CREATE TABLE PENGGUNA(
    NAMA VARCHAR(50) NOT NULL PRIMARY KEY,
    EMAIL VARCHAR(50) NOT NULL UNIQUE,
    PASSWORD VARCHAR(50) NOT NULL
);

CREATE TABLE RESEP(
    NAMA_MAKANAN VARCHAR(50) NOT NULL PRIMARY KEY,
    ASAL_NEGARA VARCHAR(50) NOT NULL,
    BAHAN_UTAMA VARCHAR(50) NOT NULL,
    LINK_TUTORIAL VARCHAR(100) NOT NULL,
    GAMBAR_MAKANAN VARCHAR(255) NOT NULL
);
